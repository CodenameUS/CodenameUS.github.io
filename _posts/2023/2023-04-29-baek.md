```c
#include <stdio.h>

void swap(int* a, int* b);

int main(void)
{
	int N, M;
	int i, j;
	scanf("%d %d", &N, &M);

	int arr[100];

	for (int n = 1; n <= N; n++) {
		arr[n] = n;
	}
	
	while (M > 0) {
		scanf("%d %d", &i, &j);

		for (int n = 0; n <= (j - i) / 2; n++) {
			swap(&arr[n + i], &arr[j - n]);
		}
		M--;
	}

	for (int n = 1; n <= N; n++) {
		printf("%d ",arr[n]);
	}
}

void swap(int* a, int* b) {
	int temp;
	temp = *b;
	*b = *a;
	*a = temp;
}
```
